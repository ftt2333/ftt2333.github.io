mat qpow(mat a, int b) {
  mat res;
  res.a[0][0] = 1, res.a[0][1] = 0;
  res.a[1][0] = 0, res.a[1][1] = 1;
  for (; b; b >>= 1, a = a * a)
    if (b & 1) res = res * a;
  return res;
}
